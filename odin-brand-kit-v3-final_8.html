/* ── DAY MODE — Warm Stone (default) ───────────────────── */
:root {
  --accent:  #7B2D45;  /* Burnt Orange — O, CTA, chips (both modes) */
  --ink:     #130E09;  /* Deep Brown   — primary text               */
  --bg:      #FAF6EE;  /* Warm White   — page background            */
  --surface: #FFFFFF;  /* White        — cards, inputs              */
  --muted:   #A89080;  /* Warm Grey    — meta, placeholders         */
  --border:  #D0BFA8;  /* Light border — dividers                   */
  --panel:   #F3EDE3;  /* Soft panel   — inner sections             */
}

/* ── DARK MODE — Deep Navy (auto · follows OS) ──────────── */
@media (prefers-color-scheme: dark) {
  :root {
    --accent:  #D4622A;  /* Burnt Orange — dark mode, pops on navy     */
    --ink:     #F0EBE3;  /* Warm Cream  — readable, not harsh white   */
    --bg:      #0B0F1A;  /* Deep Navy   — page background             */
    --surface: #1E2A3E;  /* Slate Navy  — cards, inputs               */
    --muted:   #8899AA;  /* Light Slate — meta, placeholders (light)  */
    --border:  #1C2436;  /* Dark border — dividers                    */
    --panel:   #172030;  /* Deep panel  — inner sections              */
  }
}
